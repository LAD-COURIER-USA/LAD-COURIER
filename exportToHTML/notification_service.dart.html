notification_service.dart 
1    import 'package:firebase_messaging/firebase_messaging.dart';
2    import 'package:cloud_firestore/cloud_firestore.dart';
3    import 'package:firebase_auth/firebase_auth.dart';
4    import 'package:flutter/foundation.dart';
5    import 'package:flutter_local_notifications/flutter_local_notifications.dart';
6    
7    class NotificationService {
8      final FirebaseMessaging _fcm = FirebaseMessaging.instance;
9      final FirebaseFirestore _db = FirebaseFirestore.instance;
10     final FirebaseAuth _auth = FirebaseAuth.instance;
11   
12     // Plugin para disparar las alertas visuales y sonoras
13     final FlutterLocalNotificationsPlugin _localNotifications = FlutterLocalNotificationsPlugin();
14   
15     Future<void> initialize() async {
16       // 1. Pedir permisos (Android 13+ y S24)
17       NotificationSettings settings = await _fcm.requestPermission(
18         alert: true,
19         badge: true,
20         sound: true,
21         criticalAlert: true,
22       );
23   
24       if (settings.authorizationStatus == AuthorizationStatus.authorized) {
25         debugPrint('SISTEMA LAD: Permiso concedido.');
26   
27         // 2. CONFIGURACIÓN DEL CANAL (Bunker de sonido)
28         const AndroidNotificationChannel channel = AndroidNotificationChannel(
29           'high_importance_channel', // DEBE coincidir con el de la Cloud Function
30           'Alertas de Pedidos Urgentes',
31           description: 'Este canal se usa para notificaciones que deben sonar sí o sí.',
32           importance: Importance.max,
33           playSound: true,
34           enableVibration: true,
35         );
36   
37         // Creamos el canal físicamente en el Android de Lucrecio
38         await _localNotifications
39             .resolvePlatformSpecificImplementation<AndroidFlutterLocalNotificationsPlugin>()
40             ?.createNotificationChannel(channel);
41   
42         // 3. INICIALIZACIÓN DE NOTIFICACIONES LOCALES (Para primer plano)
43         const AndroidInitializationSettings initializationSettingsAndroid =
44         AndroidInitializationSettings('@mipmap/ic_launcher_main');
45   
46         const InitializationSettings initializationSettings = InitializationSettings(
47           android: initializationSettingsAndroid,
48         );
49   
50         await _localNotifications.initialize(initializationSettings);
51   
52         // 4. EL ESCUCHA DE PRIMER PLANO (El que hace sonar el panel)
53         // Este bloque captura el mensaje cuando Lucrecio tiene el panel abierto
54         FirebaseMessaging.onMessage.listen((RemoteMessage message) {
55           RemoteNotification? notification = message.notification;
56           AndroidNotification? android = message.notification?.android;
57   
58           if (notification != null && android != null) {
59             debugPrint('SISTEMA LAD: Disparando alerta sonora en primer plano.');
60   
61             _localNotifications.show(
62               notification.hashCode,
63               notification.title,
64               notification.body,
65               NotificationDetails(
66                 android: AndroidNotificationDetails(
67                   channel.id,
68                   channel.name,
69                   channelDescription: channel.description,
70                   importance: Importance.max,
71                   priority: Priority.high,
72                   icon: android.smallIcon ?? '@mipmap/ic_launcher_main',
73                   playSound: true,
74                 ),
75               ),
76             );
77           }
78         });
79   
80         // 5. Configuración global de Firebase
81         await _fcm.setForegroundNotificationPresentationOptions(
82           alert: true,
83           badge: true,
84           sound: true,
85         );
86   
87         if (_auth.currentUser != null) {
88           await saveTokenToDatabase();
89         }
90   
91         _fcm.onTokenRefresh.listen((_) => saveTokenToDatabase());
92         _auth.authStateChanges().listen((user) {
93           if (user != null) saveTokenToDatabase();
94         });
95       }
96     }
97   
98     Future<void> saveTokenToDatabase() async {
99       final user = _auth.currentUser;
100      if (user == null) return;
101      try {
102        final String? token = await _fcm.getToken();
103        if (token != null) {
104          await _db.collection('users').doc(user.uid).update({
105            'fcmToken': token,
106            'lastTokenUpdate': FieldValue.serverTimestamp(),
107          });
108        }
109      } catch (e) {
110        debugPrint('Error guardando token: $e');
111      }
112    }
113  }
