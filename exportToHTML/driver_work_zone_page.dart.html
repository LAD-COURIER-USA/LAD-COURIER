96     Future<BitmapDescriptor> _createNumberedMarkerIconWithColor(int number, Color color) async {
97       final ui.PictureRecorder pictureRecorder = ui.PictureRecorder();
98       final Canvas canvas = Canvas(pictureRecorder);
99       final Paint paint = Paint()..color = color;
100      const double radius = 25.0;
101      canvas.drawCircle(const Offset(radius, radius), radius, paint);
102      TextPainter painter = TextPainter(textDirection: TextDirection.ltr);
103      painter.text = TextSpan(
104          text: number.toString(),
105          style: const TextStyle(fontSize: 22.0, color: Colors.white, fontWeight: FontWeight.bold)
106      );
107      painter.layout();
108      painter.paint(canvas, Offset(radius - painter.width / 2, radius - painter.height / 2));
109      final ui.Image image = await pictureRecorder.endRecording().toImage(60, 60);
110      final data = await image.toByteData(format: ui.ImageByteFormat.png);
111      return BitmapDescriptor.bytes(data!.buffer.asUint8List());
112    }

114    Future<void> _fetchLocationAndSetupStreams() async {
115      try {
116        final position = await _locationService.getCurrentLocation(context);
117        final currentUser = FirebaseAuth.instance.currentUser;
118  
119        if (currentUser != null && mounted) {
120          setState(() {
121            _driverPosition = position;
122            _fixedWorkZoneCenter = LatLng(position.latitude, position.longitude);
123          });
124  
125          _locationSubscription = Geolocator.getPositionStream(
126              locationSettings: const LocationSettings(accuracy: LocationAccuracy.high, distanceFilter: 10)
127          ).listen((newPos) {
128            if (mounted) {
129              setState(() => _driverPosition = newPos);
130              if (_activeOrders.isNotEmpty) _calculateOperationalSequence(_activeOrders);
131            }
132          });
133  
134          _userSubscription = FirebaseFirestore.instance.collection('users').doc(currentUser.uid).snapshots().listen((snapshot) {
135            if (!snapshot.exists || !mounted) return;
136            final userData = UserModel.fromFirestore(snapshot);
137            setState(() {
138              _userMaxRadius = userData.maxRadiusMiles;
139              _userMaxDropoffRadius = userData.maxDropoffRadiusMiles;
140  
141              if (userData.workZoneCenter != null) {
142                _fixedWorkZoneCenter = LatLng(userData.workZoneCenter!.latitude, userData.workZoneCenter!.longitude);
143              }
144  
145              _circles.clear();
146              _circles.add(Circle(
147                  circleId: const CircleId('pickup_zone'),
148                  center: _fixedWorkZoneCenter!,
149                  radius: _userMaxRadius * 1609.34,
150                  fillColor: Colors.deepPurple.withAlpha(25),
151                  strokeColor: Colors.deepPurple,
152                  strokeWidth: 3
153              ));
154  
155              if (_userMaxDropoffRadius > _userMaxRadius) {
156                _circles.add(Circle(
157                    circleId: const CircleId('delivery_zone'),
158                    center: _fixedWorkZoneCenter!,
159                    radius: _userMaxDropoffRadius * 1609.34,
160                    fillColor: Colors.blue.withAlpha(10),
161                    strokeColor: Colors.blueAccent.withAlpha(80),
162                    strokeWidth: 2,
163                ));
164              }
165  
166              _isLoadingLocation = false;
167              _updateNegotiatingOrdersStream(currentUser.uid);
168            });
169          });
170  
171          _activeOrdersSubscription = _orderService.getActiveOrdersStream(currentUser.uid).listen((orders) async {
172            if (!mounted) return;
173            if (orders.isEmpty) {
174              setState(() {
175                _activeOrders = [];
176                _completedPointIds.clear();
177                _activePointIdsSnapshot.clear();
178                _polylines.clear();
179                _markers.clear();
180              });
181              return;
182            }
183            Set<String> currentPointIds = {};
184            for (var o in orders) {
185              if (o.status == OrderStatus.active || o.status == OrderStatus.enRouteToPickup) currentPointIds.add("p_${o.id}");
186              if (o.status == OrderStatus.pickedUp || o.status == OrderStatus.enRouteToDelivery) currentPointIds.add("d_${o.id}");
187            }
188            for (String oldId in _activePointIdsSnapshot) {
189              if (!currentPointIds.contains(oldId)) _completedPointIds.add(oldId);
190            }
191            _activePointIdsSnapshot = currentPointIds;
192            _activeOrders = orders;
193            _calculateOperationalSequence(orders);
194          });
195        }
196      } catch (e) {
197        debugPrint("SISTEMA LAD: Error en flujo de ubicación: $e");
198        if (mounted) setState(() { _isLoadingLocation = false; });
199      }
200    }

202    void _calculateOperationalSequence(List<OrderModel> orders) {
203      List<Map<String, dynamic>> points = [];
204      if (_driverPosition == null) return;
205      for (var o in orders) {
206        if ((o.status == OrderStatus.active || o.status == OrderStatus.enRouteToPickup) && o.pickupLatLng != null) {
207          points.add({'id': "p_${o.id}", 'latlng': o.pickupLatLng, 'type': 'p', 'order': o});
208        }
209        if ((o.status == OrderStatus.pickedUp || o.status == OrderStatus.enRouteToDelivery) && o.dropoffLatLng != null) {
210          points.add({'id': "d_${o.id}", 'latlng': o.dropoffLatLng, 'type': 'd', 'order': o});
211        }
212      }
213      points.sort((a, b) {
214        double distA = Geolocator.distanceBetween(_driverPosition!.latitude, _driverPosition!.longitude, a['latlng'].latitude, a['latlng'].longitude);
215        double distB = Geolocator.distanceBetween(_driverPosition!.latitude, _driverPosition!.longitude, b['latlng'].latitude, b['latlng'].longitude);
216        return distA.compareTo(distB);
217      });
218      _drawOperationalRoute(points);
219      _buildNumberedMarkers(points);
220    }

222    void _drawOperationalRoute(List<Map<String, dynamic>> points) {
223      List<LatLng> path = [LatLng(_driverPosition!.latitude, _driverPosition!.longitude)];
224      for (var p in points) { path.add(LatLng(p['latlng'].latitude, p['latlng'].longitude)); }
225      setState(() {
226        _polylines.clear();
227        _polylines.add(Polyline(
228            polylineId: const PolylineId('op_route'),
229            points: path,
230            color: Colors.deepPurple,
231            width: 5,
232            jointType: JointType.round,
233            patterns: [PatternItem.dash(15), PatternItem.gap(10)]
234        ));
235      });
236    }

238    Future<void> _buildNumberedMarkers(List<Map<String, dynamic>> points) async {
239      final Set<Marker> newMarkers = {};
240      for (int i = 0; i < points.length; i++) {
241        final p = points[i];
242        final order = p['order'] as OrderModel;
243        int sequenceNumber = _completedPointIds.length + (i + 1);
244        newMarkers.add(Marker(
245          markerId: MarkerId(p['id']),
246          position: LatLng(p['latlng'].latitude, p['latlng'].longitude),
247          icon: await _createNumberedMarkerIconWithColor(sequenceNumber, p['type'] == 'p' ? Colors.green : Colors.red),
248          onTap: () => _openDetails(order, sequenceNumber),
249        ));
250      }
251      setState(() {
252        _markers.clear();
253        _markers.addAll(newMarkers);
254      });
255    }

257    void _openDetails(OrderModel o, int sequenceNumber) {
258      if (_driverPosition == null) return;
259      final l10n = AppLocalizations.of(context)!;
260      bool isDropoff = o.status == OrderStatus.pickedUp || o.status == OrderStatus.enRouteToDelivery;
261      String missionType = isDropoff ? l10n.deliver_label : l10n.pickup_label;
262      LatLng target = isDropoff
263          ? LatLng(o.dropoffLatLng!.latitude, o.dropoffLatLng!.longitude)
264          : LatLng(o.pickupLatLng!.latitude, o.pickupLatLng!.longitude);
265      double distanceToPoint = Geolocator.distanceBetween(_driverPosition!.latitude, _driverPosition!.longitude, target.latitude, target.longitude);
266      Navigator.push(context, MaterialPageRoute(builder: (context) => ActiveOrderDetailsPage(
267        order: o,
268        allOrders: _activeOrders,
269        missionIndex: sequenceNumber,
270        missionType: missionType,
271        isInRange: distanceToPoint <= 250,
272      )));
273    }

275    void _updateNegotiatingOrdersStream(String uid) {
276      _filteredNegotiatingOrdersStream = _orderService.getNegotiatingOrdersStream(uid).map((orders) {
277        if (_fixedWorkZoneCenter == null) return [];
278        var filtered = orders.where((o) => o.pickupLatLng != null && Geolocator.distanceBetween(_fixedWorkZoneCenter!.latitude, _fixedWorkZoneCenter!.longitude, o.pickupLatLng!.latitude, o.pickupLatLng!.longitude) <= (_userMaxRadius * 1609.34)).toList();
279        if (filtered.isEmpty && _panelController.isAttached && _panelController.size > 0.15) {
280          _panelController.animateTo(0.15, duration: const Duration(milliseconds: 400), curve: Curves.easeOut);
281        }
282        return filtered;
283      });
284      setState(() {});
285    }

287    void _triggerNewOrderAlert(String title, String body) async {
288      try {
289        HapticFeedback.vibrate();
290        HapticFeedback.heavyImpact();
291        const AndroidNotificationDetails androidPlatformChannelSpecifics = AndroidNotificationDetails(
292          'high_importance_channel', 'Alertas de Pedidos Urgentes',
293          importance: Importance.max, priority: Priority.high, playSound: true, enableVibration: true,
294        );
295        const NotificationDetails platformChannelSpecifics = NotificationDetails(android: androidPlatformChannelSpecifics);
296        await _localNotifications.show(999, title, body, platformChannelSpecifics);
297      } catch (e) {
298        debugPrint("Error al disparar alerta visual: $e");
299      }

359    Widget _buildDraggablePanel(AppLocalizations l10n) {
360      return DraggableScrollableSheet(
361        controller: _panelController,
362        initialChildSize: 0.28, minChildSize: 0.15, maxChildSize: 0.85,
363        builder: (context, scroll) => Container(
364          decoration: const BoxDecoration(
365              color: Colors.white, borderRadius: BorderRadius.vertical(top: Radius.circular(30)),
366              boxShadow: [BoxShadow(color: Colors.black26, blurRadius: 15, offset: Offset(0, -5))]
367          ),
368          child: Column(
369            children: [
370              const SizedBox(height: 12),
371              Container(width: 40, height: 5, decoration: BoxDecoration(color: Colors.grey[300], borderRadius: BorderRadius.circular(10))),
372              const SizedBox(height: 15),
373              Text(l10n.driver_work_zone_pending_requests, style: const TextStyle(color: Colors.black, fontWeight: FontWeight.w900, fontSize: 13)),
374              const SizedBox(height: 10),
375              Expanded(
376                child: StreamBuilder<List<OrderModel>>(
377                    stream: _filteredNegotiatingOrdersStream,
378                    builder: (context, snap) {
379                      final orders = snap.data ?? [];
380  
381                      if (orders.isNotEmpty && _lastKnownOrdersState.isNotEmpty) {
382                        for (var o in orders) {
383                          bool isNew = !_lastKnownOrdersState.containsKey(o.id);
384                          bool isUpdated = !isNew && o.negotiationHistory.length > _lastKnownOrdersState[o.id]!;
385  
386                          if (isNew) {
387                            _triggerNewOrderAlert(l10n.notification_new_order_title, l10n.notification_new_order_body);
388                          } else if (isUpdated) {
389                            _triggerNewOrderAlert("🚀 CONTRAOFERTA RECIBIDA", "El cliente ha respondido a tu propuesta.");
390                          }
391                        }
392                      }
393  
394                      _lastKnownOrdersState = {for (var o in orders) o.id: o.negotiationHistory.length};
395  
396                      if (orders.isEmpty) return Center(child: Text(l10n.driver_work_zone_waiting, style: const TextStyle(color: Colors.grey)));
397                      return ListView.builder(
398                          controller: scroll, padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
399                          itemCount: orders.length,
400                          itemBuilder: (context, i) {
401                            final o = orders[i];
402                            return FutureBuilder<DocumentSnapshot>(
403                                future: FirebaseFirestore.instance.collection('users').doc(o.clientId).get(),
404                                builder: (context, userSnap) {
405                                  String name = l10n.client_label;
406                                  String? photo;
407                                  if (userSnap.hasData && userSnap.data!.exists) {
408                                    final data = userSnap.data!.data() as Map<String, dynamic>;
409                                    name = data['displayName'] ?? name;
410                                    photo = data['photoURL'];
411                                  }
412                                  return _buildRequestCard(o, name, photo, l10n);
413                                }
414                            );
415                          }
416                      );
417                    }
418                ),
419              ),
420            ],
421          ),
422        ),
423      );
424    }
