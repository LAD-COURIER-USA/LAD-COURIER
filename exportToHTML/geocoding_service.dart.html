1    import 'dart:convert';
2    import 'package:http/http.dart' as http;
3    import 'package:cloud_firestore/cloud_firestore.dart';
4    import 'package:flutter/foundation.dart';
5    import 'package:lad_courier/api_keys.dart'; // Importar la clave centralizada
6    
7    class GeocodingService {
8      /// QUÉ HACE: Convierte una dirección de texto en coordenadas geográficas.
9      /// POR QUÉ: Es la única forma de obtener la inteligencia de ubicación necesaria
10     /// para calcular rutas, ETAs y confirmar entregas.
11     Future<GeoPoint?> getLatLng(String address) async {
12       // 1. Prepara la dirección para ser usada en una URL.
13       final encodedAddress = Uri.encodeComponent(address);
14       // 2. Construye la URL de la API de Geocoding usando la clave correcta.
15       final url = Uri.parse(
16           'https://maps.googleapis.com/maps/api/geocode/json?address=$encodedAddress&key=$googleMapsApiKey');
17   
18       try {
19         // 3. Ejecuta la petición a la API de Google.
20         final response = await http.get(url);
21   
22         if (response.statusCode == 200) {
23           // 4. Decodifica la respuesta JSON.
24           final data = json.decode(response.body);
25   
26           // 5. Verifica que se hayan encontrado resultados.
27           if (data['status'] == 'OK' && data['results'].isNotEmpty) {
28             // 6. Extrae las coordenadas del primer resultado.
29             final location = data['results'][0]['geometry']['location'];
30             final double lat = location['lat'];
31             final double lng = location['lng'];
32             // 7. Devuelve un GeoPoint de Firestore.
33             return GeoPoint(lat, lng);
34           } else {
35             // Imprime el error de Google para diagnóstico
36             debugPrint('Geocoding API Error: ${data['status']} - ${data['error_message']}');
37           }
38         }
39         // Si la API no responde con 200 OK o no encuentra la dirección, devuelve null.
40         return null;
41       } catch (e) {
42         // En caso de un error de red, devuelve null.
43         debugPrint('Network exception in GeocodingService: $e');
44         return null;
45       }
46     }
47   }
