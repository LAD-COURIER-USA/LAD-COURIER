deep_link_service.dart 
1    import 'dart:async';
2    import 'package:flutter/foundation.dart'; // Equipo de comunicaciones seguro.
3    import 'package:app_links/app_links.dart';
4    import 'package:shared_preferences/shared_preferences.dart';
5    
6    class DeepLinkService {
7      DeepLinkService._privateConstructor();
8      static final DeepLinkService instance = DeepLinkService._privateConstructor();
9    
10     final _appLinks = AppLinks();
11   
12     static const String _inviterIdKey = 'invitingMessengerId';
13   
14     Future<void> init() async {
15       try {
16         // === ¡MODERNIZADO! ===
17         // Se corrige la llamada a la nueva API: getInitialLink()
18         final Uri? initialUri = await _appLinks.getInitialLink();
19         if (initialUri != null) {
20           debugPrint('DeepLinkService: Enlace inicial capturado: $initialUri');
21           _processLink(initialUri);
22         }
23   
24         _appLinks.uriLinkStream.listen((Uri uri) {
25           debugPrint('DeepLinkService: Enlace de stream capturado: $uri');
26           _processLink(uri);
27         }, onError: (err) {
28           debugPrint('DeepLinkService Error: No se pudo recibir el enlace del stream: $err');
29         });
30   
31       } catch (e) {
32         debugPrint('DeepLinkService Error: Fallo al inicializar el servicio de enlaces: $e');
33       }
34     }
35   
36     Future<void> _processLink(Uri uri) async {
37       if (uri.host == 'ladcourier.com' && uri.pathSegments.length == 2 && uri.pathSegments[0] == 'invite') {
38         final inviterId = uri.pathSegments[1];
39         debugPrint('¡ID de referente capturado! ID: $inviterId');
40         final prefs = await SharedPreferences.getInstance();
41         await prefs.setString(_inviterIdKey, inviterId);
42         debugPrint('ID de referente guardado en la memoria del dispositivo.');
43       }
44     }
45   
46     static Future<String?> getStoredInviterId() async {
47       final prefs = await SharedPreferences.getInstance();
48       return prefs.getString(_inviterIdKey);
49     }
50   }
