34     Future<void> _loadUserData() async {
35       final user = FirebaseAuth.instance.currentUser;
36       if (user == null) return;
37       try {
38         final doc = await FirebaseFirestore.instance.collection('users').doc(user.uid).get();
39         if (doc.exists) {
40           _userModel = UserModel.fromFirestore(doc);
41           _nameController.text = _userModel?.displayName ?? '';
42           _phoneController.text = _userModel?.phoneNumber ?? '';
43           _addressController.text = _userModel?.mainAddress ?? '';
44           _photoUrl = _userModel?.photoURL;
45         }
46       } catch (e) {
47         debugPrint("Error: $e");
48       }
49       if (mounted) setState(() => _isLoading = false);
50     }

185    Future<void> _saveProfile() async {
186      final u = FirebaseAuth.instance.currentUser;
187      final l10n = AppLocalizations.of(context)!;
188      if (u == null) return;
189      setState(() => _isSaving = true);
190      try {
191        await FirebaseFirestore.instance.collection('users').doc(u.uid).update({'displayName': _nameController.text.trim(), 'phoneNumber': _phoneController.text.trim(), 'mainAddress': _addressController.text.trim()});
192        if (mounted) {
193          ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text(l10n.client_prof_update_success), backgroundColor: Colors.green));
194        }
195      } catch (e) {
196        if (mounted) {
197          ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text(l10n.client_prof_update_error(e.toString())), backgroundColor: Colors.red));
198        }
199      } finally {
200        if (mounted) {
201          setState(() => _isSaving = false);
202        }
203      }
204    }

206    Future<void> _changeProfilePicture() async {
207      final u = FirebaseAuth.instance.currentUser;
208      if (u == null) return;
209      final url = await _storageService.uploadProfilePicture(u.uid, context);
210      if (url != null) {
211        await FirebaseFirestore.instance.collection('users').doc(u.uid).update({'photoURL': url});
212        if (mounted) setState(() => _photoUrl = url);
213      }
214    }

216    void _switchToDriverRole() async {
217      final res = await _authService.switchUserRole(newRole: 'MESSENGER');
218      if (res == "SUCCESS" && mounted) Navigator.pop(context);
219    }
