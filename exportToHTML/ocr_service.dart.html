5    class OCRResult {
6      final String? storeName;
7      final String? fullAddress;
8      final String? streetNumber;
9      final String? zipCode;
10     final String? streetName;
11     final String? countryCode;
12   
13     OCRResult({
14       this.storeName,
15       this.fullAddress,
16       this.streetNumber,
17       this.zipCode,
18       this.streetName,
19       this.countryCode = "US",
20     });
21   }
22   
23   class OCRService {
24     final _textRecognizer = TextRecognizer(script: TextRecognitionScript.latin);
25     final _entityExtractor = EntityExtractor(language: EntityExtractorLanguage.english);
26   
27     final List<String> _commonStores = [
28       'WALMART', 'PUBLIX', 'HOME DEPOT', 'LOWES', 'TARGET', 'CVS', 'WALGREENS',
29       'ALDI', 'SEDANOS', 'PRESIDENTE', 'WINN DIXIE', 'COSTCO', '7-ELEVEN', 'WAWA',
30       'CIRCLE K', 'ROSS', 'TJ MAXX', 'MARSHALLS', 'DOLLAR TREE', 'BEST BUY'
31     ];
32   
33     final List<String> _noiseKeywords = [
34       'MANAGER', 'MGR', 'PHONE', 'TEL', 'FAX', 'ST#', 'OP#', 'TE#', 'TR#', 'TC#', 'STORE',
35       'ORDER', 'CASHIER', 'ITEMS', 'TOTAL', 'TAX', 'SUBTOTAL', 'VISA', 'MASTERCARD', 'PAGADO',
36       'DEBIT', 'CASH', 'CHANGE', 'AUTH', 'AID', 'VERIFIED', 'SALE', 'TERMINAL'
37     ];
38   
39     Future<OCRResult> analyzeReceipt(String imagePath) async {
40       try {
41         final inputImage = InputImage.fromFilePath(imagePath);
42         final RecognizedText recognizedText = await _textRecognizer.processImage(inputImage);
43   
44         String headerText = "";
45         for (var block in recognizedText.blocks.take(4)) {
46           headerText += "${block.text}\n";
47         }
48   
49         final String fullText = recognizedText.text.toUpperCase();
50         final List<String> allLines = fullText.split('\n').map((e) => e.trim()).toList();
51         final String upperHeader = headerText.toUpperCase();
52   
53         String? detectedStore;
54         String? streetNumber;
55         String? zipCode;
56         String? fullAddress;
57         String? streetName;
58   
59         for (var line in allLines.take(15)) {
60           for (var store in _commonStores) {
61             if (line.contains(store)) {
62               detectedStore = store;
63               break;
64             }
65           }
66           if (detectedStore != null) break;
67         }
68   
69         final modelManager = EntityExtractorModelManager();
70         if (!await modelManager.isModelDownloaded(EntityExtractorLanguage.english.name)) {
71           await modelManager.downloadModel(EntityExtractorLanguage.english.name);
72         }
73   
74         final List<EntityAnnotation> headerAnnotations = await _entityExtractor.annotateText(upperHeader);
75         for (var annotation in headerAnnotations) {
76           if (annotation.entities.any((e) => e.type == EntityType.address)) {
77             String text = annotation.text.toUpperCase().replaceAll('\n', ' ');
78             if (text.contains('FL') || text.contains(RegExp(r'\b3[234]\d{3}\b'))) {
79               fullAddress = text;
80               break;
81             }
82           }
83         }
84   
85         if (fullAddress != null) {
86           zipCode = RegExp(r'\b(\d{5}(?:-\d{4})?)\b').firstMatch(fullAddress)?.group(1);
87           streetNumber = RegExp(r'\b(\d{1,6})\b').firstMatch(fullAddress)?.group(1);
88           if (streetNumber != null) {
89             streetName = fullAddress.replaceFirst(streetNumber, '').trim();
90           }
91         }
92   
93         // --- LÓGICA DE ANCLA REFORZADA (DOS LÍNEAS USA) ---
94         if (zipCode == null || streetNumber == null) {
95           final RegExp zipReg = RegExp(r'\b(3[234]\d{3}(?:-\d{4})?)\b');
96           int anchorIdx = allLines.indexWhere((l) => l.contains(', FL') || zipReg.hasMatch(l));
97   
98           if (anchorIdx != -1) {
99             zipCode ??= zipReg.firstMatch(allLines[anchorIdx])?.group(1);
100            if (anchorIdx > 0) {
101              String prevLine = allLines[anchorIdx - 1];
102              final RegExp numReg = RegExp(r'^\d{1,6}\b');
103              if (numReg.hasMatch(prevLine) && !_noiseKeywords.any((k) => prevLine.contains(k))) {
104                streetNumber = numReg.firstMatch(prevLine)?.group(0);
105                streetName = prevLine.replaceFirst(streetNumber!, '').trim();
106                fullAddress = "$prevLine, ${allLines[anchorIdx]}";
107              }
108            }
109          }
110        }
111  
112        if (zipCode == null || streetNumber == null) {
113          final RegExp zipReg = RegExp(r'\b(3[234]\d{3}(?:-\d{4})?)\b');
114          final RegExp numReg = RegExp(r'^\d{1,6}\b');
115  
116          for (int i = 0; i < allLines.length && i < 20; i++) {
117            String line = allLines[i];
118            if (zipCode == null && zipReg.hasMatch(line)) zipCode = zipReg.firstMatch(line)?.group(1);
119            if (streetNumber == null && numReg.hasMatch(line)) {
120              bool isNoise = _noiseKeywords.any((k) => line.contains(k));
121              if (!isNoise && line.length > 8 && !line.contains(':') && !line.contains('#')) {
122                streetNumber = numReg.firstMatch(line)?.group(0);
123                streetName = line.replaceFirst(streetNumber!, '').trim();
124                fullAddress ??= line;
125              }
126            }
127          }
128        }
129  
130        return OCRResult(
131          storeName: detectedStore,
132          fullAddress: fullAddress,
133          streetNumber: streetNumber,
134          zipCode: zipCode,
135          streetName: streetName,
136        );
137      } catch (e) {
138        debugPrint('Error OCR Service: $e');
139        return OCRResult();
140      }
141    }
