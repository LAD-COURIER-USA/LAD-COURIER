66     void _startRadar({required int intervalSeconds}) {
67       _proximityTimer?.cancel();
68       _currentRadarInterval = intervalSeconds;
69       _proximityTimer = Timer.periodic(Duration(seconds: intervalSeconds), (timer) {
70         _checkProximity();
71       });
72     }

85     Future<void> _checkProximity() async {
86       GeoPoint? targetLatLng;
87       if (widget.order.status == OrderStatus.enRouteToPickup || widget.order.status == OrderStatus.active) {
88         targetLatLng = widget.order.pickupLatLng;
89       } else {
90         targetLatLng = widget.order.dropoffLatLng;
91       }
92   
93       if (targetLatLng != null) {
94         try {
95           Position pos = await Geolocator.getCurrentPosition(desiredAccuracy: LocationAccuracy.high);
96           double distance = Geolocator.distanceBetween(
97               pos.latitude, pos.longitude, targetLatLng.latitude, targetLatLng.longitude);
98   
99           if (mounted) {
100            setState(() {
101              _distanceToPoint = distance;
102              _isNearPoint = distance <= unlockThreshold;
103            });
104            if (distance < 600 && _currentRadarInterval != 3) {
105              _startRadar(intervalSeconds: 3);
106            } else if (distance >= 600 && _currentRadarInterval != 10) {
107              _startRadar(intervalSeconds: 10);
108            }
109          }
110        } catch (e) {
111          debugPrint("Error GPS Radar: $e");
112        }
113      }
114    }

116    Future<void> _saveProductPhotoToGallery(String url, String orderId, AppLocalizations l10n) async {
117      setState(() => _isSavingProductPhoto = true);
118      try {
119        final response = await http.get(Uri.parse(url));
120        final documentDirectory = await getTemporaryDirectory();
121        final file = File('${documentDirectory.path}/LAD_ORDER_${orderId.substring(0, 8)}.jpg');
122        file.writeAsBytesSync(response.bodyBytes);
123        await Gal.putImage(file.path, album: "LAD_CLIENT_PHOTOS");
124        if (mounted) {
125          ScaffoldMessenger.of(context).showSnackBar(SnackBar(
126            content: Text("✅ ${l10n.order_details_photo_saved}"),
127            backgroundColor: Colors.black,
128          ));
129        }
130      } catch (e) {
131        if (mounted) {
132          ScaffoldMessenger.of(context).showSnackBar(SnackBar(
133            content: Text("⚠️ ${l10n.order_details_photo_error}"),
134            backgroundColor: Colors.red,
135          ));
136        }
137      } finally {
138        if (mounted) setState(() => _isSavingProductPhoto = false);
139      }
140    }

174    Future<void> _startNavigation(GeoPoint dest, String id, bool isPickup) async {
175      final uri = Uri.parse('https://www.google.com/maps/dir/?api=1&destination=${dest.latitude},${dest.longitude}&travelmode=driving');
176      await _orderService.updateOrderStatus(
177          id, isPickup ? OrderStatus.enRouteToPickup : OrderStatus.enRouteToDelivery,
178          message: isPickup ? "🚀 Iniciando ruta de recogida." : "🚚 Iniciando ruta de entrega.");
179      if (await canLaunchUrl(uri)) await launchUrl(uri, mode: LaunchMode.externalApplication);
180    }

182    Future<void> _completeOrder(OrderModel order) async {
183      if (_deliveryPhoto == null) return;
184      setState(() => _isUploading = true);
185      try {
186        final String fileName = "delivery_${order.id}_${DateTime.now().millisecondsSinceEpoch}.jpg";
187        final ref = FirebaseStorage.instance.ref().child('delivery_evidence').child(fileName);
188  
189        await ref.putFile(_deliveryPhoto!);
190        final String downloadUrl = await ref.getDownloadURL();
191  
192        await FirebaseFirestore.instance.collection('orders').doc(order.id).update({
193          'status': OrderStatus.completed,
194          'statusMessage': '✅ ¡Pedido entregado con éxito!',
195          'completionTimestamp': Timestamp.now(),
196          'deliveryProofUrl': downloadUrl,
197        });
198  
199        await _notifyClientCompletion(order.id);
200  
201        if (mounted) Navigator.pop(context);
202      } catch (e) {
203        if (mounted) ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text("Error: $e")));
204      } finally { if (mounted) setState(() => _isUploading = false); }
205    }

207    Future<void> _notifyClientCompletion(String orderId) async {
208      if (_clientProfile?.phoneNumber == null) return;
209      final String message = "LAD COURIER: Pedido #${orderId.substring(0, 5)} entregado con éxito.";
210      final Uri smsUri = Uri.parse("sms:${_clientProfile!.phoneNumber}?body=${Uri.encodeComponent(message)}");
211      if (await canLaunchUrl(smsUri)) await launchUrl(smsUri);
212    }

435    Widget _buildActionPanel(OrderModel order, AppLocalizations l10n) {
436      bool isPickupPhase = (order.status == OrderStatus.active || order.status == OrderStatus.enRouteToPickup);
437      bool isDeliveryPhase = (order.status == OrderStatus.pickedUp || order.status == OrderStatus.enRouteToDelivery);
438  
439      return Container(
440        padding: const EdgeInsets.all(20),
441        decoration: const BoxDecoration(color: Colors.white, borderRadius: BorderRadius.vertical(top: Radius.circular(25)), boxShadow: [BoxShadow(color: Colors.black12, blurRadius: 10)]),
442        child: SafeArea(
443          child: Column(
444            mainAxisSize: MainAxisSize.min,
445            children: [
446              if (isPickupPhase && !_isNearPoint)
447                _btn(l10n.order_details_btn_go_pickup, Icons.navigation, Colors.blue, () => _startNavigation(order.pickupLatLng!, order.id, true)),
448              if (isPickupPhase && _isNearPoint)
449                _btn(l10n.order_details_btn_arrived, Icons.check_box, Colors.green, () async {
450  
451                  // 🚀 ENGRANAJE DE AUTO-APRENDIZAJE SOBERANO REPARADO
452                  try {
453                    final pos = await Geolocator.getCurrentPosition(desiredAccuracy: LocationAccuracy.high);
454                    final String? driverId = FirebaseAuth.instance.currentUser?.uid;
455  
456                    if (driverId != null) {
457                      final String fullAddr = order.pickupAddress.toUpperCase();
458                      // Extraer número (primer bloque de dígitos)
459                      final String streetNum = fullAddr.split(' ')[0].replaceAll(RegExp(r'\D'), '');
460                      // Regex Flexible: Acepta ZIP de 5 o 5+4 dígitos (Florida 32, 33, 34)
461                      final RegExp zipRegex = RegExp(r'\b(\d{5}(?:-\d{4})?)\b');
462                      final String? zip = zipRegex.firstMatch(fullAddr)?.group(0);
463  
464                      String store = "Comercio Local";
465                      if (order.packageDetails != null && order.packageDetails!.contains("RECOGER EN ")) {
466                        final parts = order.packageDetails!.split("RECOGER EN ");
467                        if (parts.length > 1) {
468                          store = parts[1].split(".")[0].trim();
469                        }
470                      }
471  
472                      if (zip != null && streetNum.isNotEmpty) {
473                        await _geodataService.registerNewValidatedStore(
474                          zip: zip,
475                          streetNumber: streetNum,
476                          storeName: store,
477                          fullAddress: order.pickupAddress,
478                          lat: pos.latitude,
479                          lng: pos.longitude,
480                          driverId: driverId,
481                        );
482                        debugPrint("LAD: Inteligencia Soberana alimentada.");
483                      }
484                    }
485                  } catch (e) {
486                    debugPrint("LAD: Error en Auto-Aprendizaje: $e");
487                  }
488                  // -------------------------------------------
489  
490                  await _orderService.updateOrderStatus(order.id, OrderStatus.pickedUp, message: "📦 Paquete recogido.");
491                  if (mounted) Navigator.pop(context);
492                }),
493              if (isDeliveryPhase && !_isNearPoint)
494                _btn(l10n.order_details_btn_go_delivery, Icons.directions_car, Colors.deepPurple, () => _startNavigation(order.dropoffLatLng!, order.id, false)),
495              if (isDeliveryPhase && _isNearPoint) ...[
496                if (_deliveryPhoto == null)
497                  _btn(l10n.order_details_btn_photo, Icons.camera_alt, Colors.orange, () => _takePhoto(l10n))
498                else
499                  _btn(l10n.order_details_btn_finish, Icons.verified, Colors.green, () => _completeOrder(order)),
500              ],
501            ],
502          ),
503        ),
504      );
505    }
