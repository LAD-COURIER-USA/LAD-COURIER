97     void _triggerDynamicFiltering() {
98       if (_debounce?.isActive ?? false) {
99         _debounce?.cancel();
100      }
101      _debounce = Timer(const Duration(milliseconds: 1500), () async {
102        if (!mounted) {
103          return;
104        }
105  
106        final String pickupText = _pickupController.text;
107        if (pickupText.length > 5 && _validatedPickupLatLng == null) {
108          final zipMatch = RegExp(r'\b(3[234]\d{3})\b').firstMatch(pickupText);
109          final numMatch = RegExp(r'\d+').firstMatch(pickupText);
110  
111          bool foundInGeodata = false;
112          if (zipMatch != null && numMatch != null) {
113            final String zip = zipMatch.group(1)!;
114            final String num = numMatch.group(0)!;
115  
116            debugPrint("LAD: Iniciando búsqueda en Firestore -> ZIP: $zip, NUM: $num");
117  
118            final store = await _geodataService.findStoreByDna(
119                zip: zip,
120                streetNumber: num
121            );
122  
123            if (store != null && mounted) {
124              foundInGeodata = true;
125              debugPrint("LAD: ¡MATCH ENCONTRADO EN FIRESTORE! Ahorro de API ejecutado.");
126              setState(() {
127                _isPickupVerified = true;
128                _validatedPickupLatLng = GeoPoint(store['gps']['lat'], store['gps']['lon']);
129                _validatedStoreAddress = pickupText;
130              });
131            } else {
132              debugPrint("LAD: No se encontró el local $num en el ZIP $zip en tu base geodata_fl.");
133            }
134          }
135  
136          if (!foundInGeodata) {
137            debugPrint("LAD: Usando respaldo -> Geocoding Externo (Google)...");
138            final latLng = await _geocodingService.getLatLng(pickupText);
139            if (mounted && latLng != null) {
140              setState(() {
141                _isPickupVerified = false;
142                _validatedPickupLatLng = latLng;
143                _validatedStoreAddress = pickupText;
144              });
145            }
146          }
147        }
148  
149        if (_dropoffController.text.length > 5 && _validatedDropoffLatLng == null) {
150          final latLng = await _geocodingService.getLatLng(_dropoffController.text);
151          if (mounted && latLng != null) {
152            setState(() {
153              _validatedDropoffLatLng = latLng;
154              _validatedDropoffAddress = _dropoffController.text;
155            });
156          }
157        }
158      });
159    }

196    Future<void> _pickProductPhoto() async {
197      setState(() => _isUploadingPhoto = true);
198  
199      final url = await _storageService.uploadProductPhoto(
200          "prod_${DateTime.now().millisecondsSinceEpoch}",
201          context,
202          onLocalPathPicked: (path) async {
203            final ocrResult = await _ocrService.analyzeReceipt(path);
204  
205            Map<String, dynamic>? validatedStore;
206            if (ocrResult.zipCode != null && ocrResult.streetNumber != null) {
207              validatedStore = await _geodataService.findStoreByDna(
208                  zip: ocrResult.zipCode!,
209                  streetNumber: ocrResult.streetNumber!,
210                  countryCode: ocrResult.countryCode ?? "US"
211              );
212            }
213  
214            if (mounted) {
215              _showOcrSuggestions(ocrResult, validatedStore);
216            }
217          }
218      );
219  
220      if (!mounted) {
221        return;
222      }
223      if (url != null) setState(() => _productPhotoUrl = url);
224      setState(() => _isUploadingPhoto = false);
225    }

320    void _handleOrderAction() async {
321      if (!(_formKey.currentState?.validate() ?? false)) {
322        ScaffoldMessenger.of(context).showSnackBar(const SnackBar(content: Text("⚠️ Por favor, completa los campos obligatorios."), backgroundColor: Colors.redAccent));
323        return;
324      }
325  
326      setState(() => _isLoading = true);
327      try {
328        final pickup = _validatedPickupLatLng ?? await _geocodingService.getLatLng(_pickupController.text);
329        final dropoff = _validatedDropoffLatLng ?? await _geocodingService.getLatLng(_dropoffController.text);
330  
331        if (!mounted) {
332          return;
333        }
334        if (pickup == null || dropoff == null) {
335          throw Exception("No pudimos localizar las direcciones. Verifica que sean correctas.");
336        }
337  
338        if (_currentMessenger == null) {
339          Navigator.push(context, MaterialPageRoute(builder: (context) => DriverSelectionPage(
340            pickupLatLng: pickup,
341            dropoffLatLng: dropoff,
342            pickupAddress: _pickupController.text,
343            dropoffAddress: _dropoffController.text,
344            serviceType: _selectedService,
345            packageDetails: _descriptionController.text,
346            productPhotoUrl: _productPhotoUrl,
347          )));
348        } else {
349          await _createOrderDirect(pickup, dropoff);
350        }
351      } catch (e) {
352        if (mounted) {
353          ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text("Error: $e"), backgroundColor: Colors.red));
354        }
355      } finally {
356        if (mounted) {
357          setState(() => _isLoading = false);
358        }
359      }
360    }

362    Future<void> _createOrderDirect(GeoPoint p, GeoPoint d) async {
363      final user = FirebaseAuth.instance.currentUser;
364      if (_currentMessenger == null) {
365        return;
366      }
367      await _orderService.createOrder(
368        clientId: user!.uid, clientName: user.displayName ?? 'Cliente', clientPhotoUrl: user.photoURL,
369        assignedMessengerId: _currentMessenger!['id'], messengerName: _currentMessenger!['name'],
370        messengerPhotoUrl: _currentMessenger!['photoURL'], serviceType: _selectedService,
371        pickupAddress: _pickupController.text, pickupLatLng: p, dropoffAddress: _dropoffController.text,
372        dropoffLatLng: d, packageDetails: _descriptionController.text, productPhotoUrl: _productPhotoUrl,
373      );
374      if (mounted) {
375        Navigator.pop(context);
376        ScaffoldMessenger.of(context).showSnackBar(const SnackBar(content: Text("🚀 ¡ORDEN ENVIADA EXITOSAMENTE!"), backgroundColor: Colors.green));
377      }
378    }

396    List<UserModel> get _filteredMessengers {
397      debugPrint("LAD: Iniciando filtrado de drivers para servicio: $_selectedService");
398      return _linkedMessengers.where((m) {
399        if (!m.isMessengerActive) {
400          return false;
401        }
402  
403        final messengerServices = m.availableServices.map((s) => s.toString().toLowerCase()).toList();
404        final selected = _selectedService.toLowerCase();
405  
406        // 1. Intento por ID directo (courier, shopping, logistics)
407        bool hasService = messengerServices.contains(selected);
408  
409        // 2. Si no lo encuentra por ID, buscamos por palabras clave (Políglota - 5 idiomas)
410        if (!hasService) {
411          if (selected == 'courier') {
412            hasService = messengerServices.any((s) =>
413            s.contains('paquet') || s.contains('courier') || s.contains('coursier') ||
414                s.contains('correio') || s.contains('mesaj') || s.contains('kouriyé'));
415          } else if (selected == 'shopping') {
416            hasService = messengerServices.any((s) =>
417            s.contains('compra') || s.contains('shop') || s.contains('achat') ||
418                s.contains('encargo') || s.contains('acha') || s.contains('komisyon'));
419          } else if (selected == 'logistics') {
420            hasService = messengerServices.any((s) => s.contains('logist'));
421          }
422        }
423  
424        if (!hasService) {
425          debugPrint("LAD: Driver ${m.displayName} descartado: No ofrece $selected");
426          return false;
427        }
428  
429        if (m.workZoneCenter != null) {
430          double radiusLimit = 15.0;
431          final plan = (m.subscriptionType ?? 'lite').toLowerCase();
432          if (plan == 'standard' || plan == 'standart') {
433            radiusLimit = 25.0;
434          } else if (plan == 'pro') {
435            radiusLimit = 120.0;
436          }
437  
438          if (_validatedPickupLatLng != null) {
439            double distP = Geolocator.distanceBetween(
440                m.workZoneCenter!.latitude, m.workZoneCenter!.longitude,
441                _validatedPickupLatLng!.latitude, _validatedPickupLatLng!.longitude
442            ) / 1609.34;
443  
444            if (distP > radiusLimit) {
445              debugPrint("LAD: Driver ${m.displayName} descartado: Pickup fuera de rango (${distP.toStringAsFixed(1)} mi)");
446              return false;
447            }
448          }
449  
450          if (_validatedDropoffLatLng != null) {
451            double distD = Geolocator.distanceBetween(
452                m.workZoneCenter!.latitude, m.workZoneCenter!.longitude,
453                _validatedDropoffLatLng!.latitude, _validatedDropoffLatLng!.longitude
454            ) / 1609.34;
455  
456            if (distD > radiusLimit) {
457              debugPrint("LAD: Driver ${m.displayName} descartado: Dropoff fuera de rango (${distD.toStringAsFixed(1)} mi)");
458              return false;
459            }
460          }
461        } else {
462          debugPrint("LAD: Driver ${m.displayName} sin zona definida. Se oculta por seguridad.");
463          return false;
464        }
465        return true;
466      }).toList();
467    }

607    Widget _buildServiceSelector(AppLocalizations l10n) {

618          children: serviceOptions.entries.map((entry) {
619            final sKey = entry.key;
620            final sLabel = entry.value;

622            return GestureDetector(

640                ),
641                child: Row(
642                  children: [

698      Color fillColor = Colors.white;
699      Color borderColor = Colors.grey[100]!;
700

702        if (_isPickupVerified) {

706          fillColor = Colors.amber[50]!;
707          borderColor = Colors.amber[300]!;

709      }

716              color: fillColor,

720            ),
721            child: TextFormField(

723              maxLines: lines,
724              style: const TextStyle(fontWeight: FontWeight.w900, fontSize: 15, color: Colors.black),
725              decoration: InputDecoration(
