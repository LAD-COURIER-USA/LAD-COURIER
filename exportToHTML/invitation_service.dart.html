1    import 'package:firebase_auth/firebase_auth.dart';
2    import 'package:flutter/material.dart';
3    import 'package:share_plus/share_plus.dart';
4    import 'package:lad_courier/models/user_model.dart';
5    import 'package:lad_courier/l10n/app_localizations.dart';
6    
7    /// Un servicio dedicado para manejar toda la lógica relacionada con
8    /// la creación y gestión de invitaciones.
9    class InvitationService {
10     /// Construye un enlace de invitación personal para el usuario actual (Referido estándar)
11     void shareInvitationLink(BuildContext context) {
12       final user = FirebaseAuth.instance.currentUser;
13       final l10n = AppLocalizations.of(context)!;
14   
15       if (user == null) {
16         if (context.mounted) {
17           ScaffoldMessenger.of(context).showSnackBar(
18             SnackBar(
19               content: Text(l10n.service_invitation_error_user),
20               backgroundColor: Colors.red,
21             ),
22           );
23         }
24         return;
25       }
26   
27       final String userId = user.uid;
28       const String domain = "ladcourier.com";
29       final String invitationLink = 'https://$domain/invite?id=$userId&type=referral';
30   
31       final String shareMessage = l10n.service_invitation_share_msg(invitationLink);
32   
33       SharePlus.instance.share(
34         ShareParams(
35           text: shareMessage,
36           subject: l10n.service_invitation_subject,
37         ),
38       );
39     }
40   
41     /// NUEVO MÉTODO TÁCTICO: Recomendar un mensajero específico por parte de un cliente.
42     /// Incluye banderas para evitar que cuente como red de crecimiento del mensajero.
43     void shareMessengerRecommendation({
44       required BuildContext context,
45       required UserModel messenger,
46       required UserModel client,
47     }) {
48       final l10n = AppLocalizations.of(context)!;
49       const String domain = "ladcourier.com";
50   
51       // El enlace incluye el ID del cliente (quien invita), el ID del mensajero (el recomendado)
52       // y el tipo 'recommendation' para que el sistema sepa que no es captación directa del mensajero.
53       final String invitationLink =
54           'https://$domain/invite?id=${client.uid}&recommendedMessengerId=${messenger.uid}&type=recommendation';
55   
56       final String shareMessage = l10n.service_recommend_share_msg(
57         messenger.displayName ?? "Driver",
58         invitationLink,
59       );
60   
61       SharePlus.instance.share(
62         ShareParams(
63           text: shareMessage,
64           subject: l10n.service_recommend_subject,
65         ),
66       );
67     }
68   }
