50     Future<void> _initializeNotifications() async {
51       const AndroidInitializationSettings initializationSettingsAndroid =
52       AndroidInitializationSettings('ic_launcher_driver');
53       const InitializationSettings initializationSettings = InitializationSettings(android: initializationSettingsAndroid);
54       await _localNotifications.initialize(initializationSettings);
55     }

57     Future<void> _fetchDriverProfile() async {
58       final currentUser = FirebaseAuth.instance.currentUser;
59       if (currentUser != null) {
60         final profile = await _userService.getUser(currentUser.uid);
61         if (mounted) {
62           setState(() {
63             _driverProfile = profile;
64             _isLoadingProfile = false;
65           });
66           if (profile?.isMessengerActive ?? false) {
67             _startGlobalOrderListener(profile!.uid);
68           }
69         }
70       }
71     }
72   
73     void _startGlobalOrderListener(String uid) {
74       _globalOrdersSubscription?.cancel();
75       _globalOrdersSubscription = _orderService.getNegotiatingOrdersStream(uid).listen((orders) {
76         if (!mounted) return;
77         final l10n = AppLocalizations.of(context);
78         if (l10n == null) return;
79   
80         if (_lastKnownOrdersState.isNotEmpty) {
81           for (var o in orders) {
82             bool isNew = !_lastKnownOrdersState.containsKey(o.id);
83             bool isUpdated = !isNew && o.negotiationHistory.length > _lastKnownOrdersState[o.id]!;
84   
85             if (isNew) {
86               _triggerNotification(
87                   l10n.notification_new_order_title,
88                   l10n.notification_new_order_body
89               );
90             } else if (isUpdated && o.lastPriceOfferedBy == 'client') {
91               _triggerNotification(
92                   "🚀 CONTRAOFERTA RECIBIDA",
93                   "El cliente ${o.clientName} ha respondido a tu propuesta."
94               );
95             }
96           }
97         }
98         _lastKnownOrdersState = {for (var o in orders) o.id: o.negotiationHistory.length};
99       });
100    }

102    void _triggerNotification(String title, String body) async {
103      HapticFeedback.vibrate();
104      const AndroidNotificationDetails androidPlatformChannelSpecifics = AndroidNotificationDetails(
105        'global_orders_channel', 'Notificaciones Globales de Órdenes',
106        importance: Importance.max, priority: Priority.high, playSound: true, enableVibration: true,
107      );
108      const NotificationDetails platformChannelSpecifics = NotificationDetails(android: androidPlatformChannelSpecifics);
109      await _localNotifications.show(DateTime.now().millisecond, title, body, platformChannelSpecifics);
110    }

112    String _getTranslatedServiceName(BuildContext context, String serviceId) {
113      final l10n = AppLocalizations.of(context)!;
114      switch (serviceId) {
115        case 'Paquetería y Mensajería':
116          return l10n.driver_service_courier;
117        case 'Logística Especializada':
118          return l10n.driver_service_logistics;
119        case 'Compras y Encargos':
120          return l10n.driver_service_shopping;
121        default:
122          return serviceId;
123      }
124    }

126    Future<void> _toggleStatus(bool isGoingOnline) async {
127      if (_driverProfile == null) return;
128  
129      final l10n = AppLocalizations.of(context)!;
130      final navigator = Navigator.of(context);
131  
132      if (isGoingOnline) {
133        if (_driverProfile!.photoURL == null || _driverProfile!.photoURL!.isEmpty) {
134          _showErrorDialog(l10n.driver_error_no_photo, l10n.driver_error_no_photo_msg);
135          return;
136        }
137        if (_driverProfile!.displayName == null || _driverProfile!.displayName!.isEmpty ||
138            _driverProfile!.phoneNumber == null || _driverProfile!.phoneNumber!.isEmpty) {
139          _showErrorDialog(l10n.driver_error_incomplete_data, l10n.driver_error_incomplete_data_msg);
140          return;
141        }
142        if (_driverProfile!.vehicleDescription == null || _driverProfile!.vehicleDescription!.isEmpty) {
143          _showErrorDialog(l10n.driver_error_no_vehicle, l10n.driver_error_no_vehicle_msg);
144          return;
145        }
146        if (!_bypassPaymentForTesting) {
147          if (_driverProfile!.subscriptionType == null || _driverProfile!.subscriptionStatus != 'active') {
148            _showErrorDialog(l10n.driver_error_no_membership, l10n.driver_error_no_membership_msg);
149            return;
150          }
151        }
152        if (_driverProfile!.availableServices.isEmpty) {
153          _showErrorDialog(l10n.driver_error_no_services, l10n.driver_error_no_services_msg);
154          return;
155        }
156      } else {
157        final activeOrders = await _orderService.getActiveOrdersOnce(_driverProfile!.uid);
158  
159        if (!mounted) return;
160  
161        if (activeOrders.isNotEmpty) {
162          _showSnackBar(l10n.driver_active_missions_alert, Colors.red);
163          return;
164        }
165        _globalOrdersSubscription?.cancel();
166        _lastKnownOrdersState.clear();
167      }
168  
169      if (!mounted) return;
170      setState(() { _isToggleLoading = true; });
171  
172      try {
173        await _userService.updateMessengerActiveStatus(_driverProfile!.uid, isGoingOnline, context);
174  
175        if (!mounted) return;
176  
177        setState(() {
178          _driverProfile = _driverProfile!.copyWith(isMessengerActive: isGoingOnline);
179        });
180  
181        if (isGoingOnline) {
182          _startGlobalOrderListener(_driverProfile!.uid);
183          navigator.push(MaterialPageRoute(builder: (context) => const DriverWorkZonePage()));
184        }
185      } catch (e) {
186        if (mounted) _showSnackBar('Error: $e', Colors.red);
187      } finally {
188        if (mounted) setState(() { _isToggleLoading = false; });
189      }
190    }

349    void _showServicesDialog(BuildContext context) {
350      if (_driverProfile == null) return;
351      final l10n = AppLocalizations.of(context)!;
352      final List<String> temp = List<String>.from(_driverProfile!.availableServices);
353      showDialog(
354          context: context,
355          builder: (c) => StatefulBuilder(builder: (c, setS) => AlertDialog(
356            backgroundColor: Colors.white,
357            title: Text(l10n.driver_dialog_services_title, style: const TextStyle(fontWeight: FontWeight.w900, color: Colors.black)),
358            content: Column(
359              mainAxisSize: MainAxisSize.min,
360              children: _allServices.map((s) => CheckboxListTile(
361                  title: Text(_getTranslatedServiceName(context, s),
362                      style: const TextStyle(fontWeight: FontWeight.w900, color: Colors.black, fontSize: 14)),
363                  value: temp.contains(s),
364                  activeColor: Colors.black,
365                  onChanged: (v) => setS(() => v! ? temp.add(s) : temp.remove(s))
366              )).toList(),
367            ),
368            actions: [
369              FilledButton(
370                  style: FilledButton.styleFrom(backgroundColor: Colors.black),
371                  onPressed: () async {
372                    await _userService.updateAvailableServices(_driverProfile!.uid, temp);
373                    _fetchDriverProfile();
374                    if (c.mounted) Navigator.pop(c);
375                  },
376                  child: Text(l10n.driver_btn_confirm, style: const TextStyle(color: Colors.greenAccent, fontWeight: FontWeight.bold))
377              )
378            ],
379          ))
380      );
381    }
