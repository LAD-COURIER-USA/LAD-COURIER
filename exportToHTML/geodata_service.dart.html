4    class GeodataService {
5      final FirebaseFirestore _db = FirebaseFirestore.instance;
6      final Map<String, Map<String, dynamic>> _memoryCache = {};
7    
8      /// Busca un local por su "DNA" (ZIP + Número de calle)
9      /// Optimizado para evitar errores de índices de Firestore mediante filtrado en cliente.
10     Future<Map<String, dynamic>?> findStoreByDna({
11       required String zip,
12       required String streetNumber,
13       String countryCode = "US",
14     }) async {
15       if (zip.isEmpty || streetNumber.isEmpty) return null;
16   
17       final cleanZip = zip.trim();
18       final cleanNum = streetNumber.trim();
19   
20       final cacheKey = "${countryCode}_${cleanZip}_$cleanNum";
21       if (_memoryCache.containsKey(cacheKey)) return _memoryCache[cacheKey];
22   
23       try {
24         String collectionName = countryCode == "US" ? 'geodata_fl' : 'geodata_${countryCode.toLowerCase()}';
25   
26         // 🔍 ESTRATEGIA FAILSAFE: Solo buscamos por número para evitar el error de "Index Required"
27         // Traemos los locales que coincidan con el número de calle.
28         final query = await _db
29             .collection(collectionName)
30             .where('address.number', isEqualTo: cleanNum)
31             .limit(40) 
32             .get();
33   
34         if (query.docs.isNotEmpty) {
35           // 🎯 FILTRADO EN EL TELÉFONO: Buscamos el ZIP correcto entre los resultados obtenidos
36           final matches = query.docs.where((doc) {
37             final dbZip = doc.data()['address']?['zip']?.toString() ?? "";
38             return dbZip.startsWith(cleanZip);
39           });
40   
41           if (matches.isNotEmpty) {
42             final data = matches.first.data();
43             
44             if (data['address'] != null) {
45               final addr = data['address'];
46               data['address']['full'] = "${addr['number']} ${addr['street']}, ${addr['city']}, ${addr['state']} ${addr['zip']}".toUpperCase();
47             }
48   
49             _memoryCache[cacheKey] = data;
50             return data;
51           }
52         }
53         return null;
54       } catch (e) {
55         debugPrint("LAD: Error en GeodataService (findStoreByDna): $e");
56         return null;
57       }
58     }
59   
60     Future<void> registerNewValidatedStore({
61       required String? zip,
62       required String streetNumber,
63       required String storeName,
64       required String fullAddress,
65       required double lat,
66       required double lng, 
67       required String driverId,
68     }) async {
69       if (zip == null || streetNumber.isEmpty) return;
70       try {
71         String collectionName = 'geodata_fl';
72         List<String> parts = fullAddress.split(',');
73         String street = streetNumber;
74         String city = "FLORIDA";
75         String state = "FL";
76   
77         if (parts.length >= 3) {
78           street = parts[0].trim();
79           city = parts[1].trim();
80           state = parts[2].trim().split(' ')[0];
81         }
82   
83         final docId = "US_${state}_${zip}_$streetNumber".replaceAll(' ', '_').toUpperCase();
84   
85         await _db.collection(collectionName).doc(docId).set({
86           'id': docId,
87           'name': storeName.toUpperCase(),
88           'active': true,
89           'category': "NEGOCIO",
90           'is_verified': true,
91           'last_updated': FieldValue.serverTimestamp(),
92           'source': 'user_validated',
93           'validated_by': driverId,
94           'address': {
95             'number': streetNumber,
96             'street': street.toUpperCase(),
97             'city': city.toUpperCase(),
98             'state': state.toUpperCase(),
99             'zip': zip,
100            'country': "US",
101          },
102          'gps': {
103            'lat': lat,
104            'lon': lng,
105          },
106          'search_key': "US_${state}_${zip}_$streetNumber",
107        }, SetOptions(merge: true));
108        clearCache();
109      } catch (e) {
110        debugPrint("Error al registrar tienda validada: $e");
111      }
112    }
113  
114    void clearCache() => _memoryCache.clear();
115  }
